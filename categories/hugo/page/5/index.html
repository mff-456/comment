mff的个人博客

April 28, 2020 mff

1 minute read

npm install vue-i18n

main.js文件中注册

// 导入多语言包
import VueI18n from 'vue-i18n'

// 注册
Vue.use(VueI18n);

// 导入语言包
const i18n = new VueI18n({

  // 当前默认语言
  locale:'en',
  // 语言包声明
  messages:{
    // 导入具体文件(这里的文件需手动创建)
    'zh':require('./lang/zh'),
    'en':require('./lang/en')
  }

});
