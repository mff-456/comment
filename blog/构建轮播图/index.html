后端(models.py新建模型carousel表)

# 创建轮播表

class Carousel(Base):
    name = models.CharField(max_length=200)
    src = models.CharField(max_length=200)
    img = models.CharField(max_length=200)
    # 声明表名

    class Meta:
        db_table = "carousel"

执行迁移命令

python manage.py makemigrations myapp
python manage.py migrate

生成数据库后暂时添加几条数据用于测试

新建序列化操作文件(myser.py)

#导包

from rest_framework import serializers
# 导入需要序列化的表

from myapp.models import Carousel
# 建立序列化类


class CarouselSer(serializers.ModelSerializer):

    class Meta:
        model = Carousel
        fields = "__all__"

后端创建轮播图接口(GetCarousel)

from rest_framework.views import APIView
from myapp.models import Carousel
from myapp.myser import CarouselSer
from rest_framework.response import Response

# 轮播图后台接口

class GetCarousel(APIView):
    def get(self,request):
        # 读库

        carousels = Carousel.objects.all()
        # 序列化操作

        carousels_ser = CarouselSer(carousels,many=True)
        return Response(carousels_ser.data)
