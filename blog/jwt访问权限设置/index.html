jwt(json web token)

jwt是前端发送请求向后端获取数据时添加了一个token验证，当用户操作合法时才会获取到token，没有权限时访问不到后端接口数据

1. 流程图

2. demo

  1. 下载jwt

    pip install jwt

  2. 生成jwt

    import jwt
import datetime
# 载荷中加入生命周期概念

playload = {
    # 过期时间 (当前时间 + 设置的时间).timestamp(强转时间戳)

    'exp' : int((datetime.datetime.now() + datetime.timedelta(seconds=30)).timestamp()),
    'data':{'uid':2}
}
# 生成jwt(第一个参数为需要生成token的参数，第二个为密钥(自己设定)，第三个为固定参数(算法))

encode_jwt = jwt.encode(playload,'mff-project',algorithm='HS256')
# 转码

encode_str = str(encode_jwt,'utf-8')
print(encode_str)

  3. jwt接口测试

    # 测试jwt接口

class Jwt_test(APIView):
    def get(self,request):
        # 接收前端的用户id

        uid = request.GET.get('uid',None)
        print(uid)
        # 接收前端的token

        myjwt = request.GET.get('jwt',None)
        print(myjwt)
        # 验证用户合法性

        try:
            # 解密上传的jwk(这里的密钥要与生成的密钥一致)

            decode_jwt = jwt.decode(myjwt,'mff-project',algorithms=['HS256'])
            print(decode_jwt['data']['uid'])
        except Exception as e:
               
            return Response({'code':403,'message':'无权访问'})
   
        # 判断上传的id与生成并解密的id是否一致

        if int(uid) != int(decode_jwt['data']['uid']):
   
            return Response({'code':403,'message':'密钥无权限'})
     # jwt正确后查询数据库数据

        user = User.objects.get(id=int(uid))
     # 向前端返回数据

        return Response({'uid':uid,'username':user.username,'code':200,'message':'成功'})

  4. 前端测试(jwt_test.vue)

3. 用法总结

import jwt
import datetime

# 载荷中加入生命周期概念

playload = {
    # 过期时间 (当前时间 + 设置的时间).timestamp(强转时间戳)

    'exp' : int((datetime.datetime.now() + datetime.timedelta(seconds=30)).timestamp()),
    'data':{'uid':2}
}
# 生成jwt

encode_jwt = jwt.encode(playload,'mff-project',algorithm='HS256')

# 转码

encode_str = str(encode_jwt,'utf-8')

# 后端解密

decode_jwt = jwt.decode(encode_str,'mff-project',algorithms=['HS256'])

print(encode_str)
print(decode_jwt['data']['uid'])
