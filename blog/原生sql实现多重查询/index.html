在进行模糊查询后会发现满足不了一些必要的条件，他只能查询一个关键字的参数，这时候就得用多重查询来实现多个关键字的查询方法

demo

前端(vue.js)接收参数时的数据类型及转换

  1. 首先创建好vue文件(search.vue)记得配置好路由

    <template>
  <div>
      <!-- 搜索框 -->
      <Search @search="search" v-model="text"></Search>
      <table>
          <!-- 遍历数据 -->
          <tr v-for="(item,index) in goodslist" :key="index">
              <td>
                  <video width="250" height="250" :src="item.video" controls="controls"></video>
              </td>

                  <h4>
                      <div v-html="item.name"></div>
                  </h4>
              </td>
              <td>
                  <span class="emphasis">{{ item.price }}</span>
              </td>
          </tr>
      </table>
  </div>
</template>
   
<script>
export default {
    data(){
        return{
            text:'',
            // 商品列表
            goodslist:[],
        }
    },
    mounted:function(){
   
    },
    methods:{
        // 获取关键字商品信息
        search:function(){    
            console.log(this.text);
            this.axios.get('http://127.0.0.1:8000/search/',{params:{text:this.text}}).then((result)=>{

  2. 后端(django)接口完成多重查询功能

    from rest_framework.views import APIView
#导入原生sql模块

from django.db import connection
from rest_framework.response import Response
   
# 格式化结果集

def dictfetch(cursor):
    # 声明描述符

    desc = cursor.description
   
    # 行转列操作(原数据为tuple格式,转换为键值对压缩到列表中提供给前端使用)

   
    return [ dict( zip([col[0] for col in desc],row )) 
   
        for row in cursor.fetchall()
       
    ]
   
# 多重查询接口

class Search(APIView):
    def get(self,request):
        # 检索字段

        text = request.GET.get('text',None)
   
        # 判断空格

        if ' ' in text:
            text = text.split(" ")
        # 判断中文逗号

        elif '，' in text:
            text = text.split("，")
        # 判断英文逗号

        elif ',' in text:
            text = text.split(",")
        else:
            text = [text]
   
        # 动态拼接

        sql = ""
        for val in text:
   
            sql += "or name like '%%%s%%'" % val
           
        # 去掉开头的"or"

        sql = sql.lstrip('or')
   
        # 建立游标对象

        cursor = connection.cursor()
   
        # 执行sql

        cursor.execute("select name,id,img,price,video from goods where id!=0 and (" + sql + ")")
   
        # 查询(查询到的数据为tuple格式，前端无法读取)

        # result = cursor.fetchall()

   
        # 调用格式化结果集的方法

        result = dictfetch(cursor)
   
        if result == []:
            return Response({'code':-1})
   
        return Response({'data':result})
