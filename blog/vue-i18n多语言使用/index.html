vue-i18n多语言使用

mff的个人博客

April 28, 2020 mff

1 minute read

  • Twitter

  • Google+

  • Facebook

  • Reddit

  • LinkedIn

npm安装

npm install vue-i18n

main.js文件中注册

// 导入多语言包
import VueI18n from 'vue-i18n'

// 注册
Vue.use(VueI18n);

// 导入语言包
const i18n = new VueI18n({

  // 当前默认语言
  locale:'en',
  // 语言包声明
  messages:{
    // 导入具体文件(这里的文件需手动创建)
    'zh':require('./lang/zh'),
    'en':require('./lang/en')
  }

});

new Vue({
  el: '#app',
  router,
  // 这里记得引入i18n
  i18n,
  components: { App },
  template: '<App/>',
  render: h => h(App)
})

创建lang文件夹(在main.js文件的同级目录(src)创建)

  1. 在lang文件夹下创建(zh.js)

    // 双语规范的变量(简体中文)
   
export const m = {
   
    'index':'美多商城'
   
}

  2. 同理在lang文件夹下创建(en.js)

    // 双语规范(英文)
   
export const m = {
   
    'index':'Mei Duo Shop'
   
}

设置默认语言(在需要的页面钩子方法中)

<template>
	<div>
        <!-- 固定写法$t(调用js文件里变量为m的数据) -->
        <p>{{ $t('m.index') }}</p>
    </div>
</template>
<script>
    export default {
        mounted:function(){
            // 查询浏览器使用语言
            if (navigator.language == 'zh-CN'){
                // 设置中文
                this.$i18n.locale = 'zh';
            }else{
                this.$i18n.locale = 'en';
            }
        }
    }
</script>

做一个中英文切换按钮

<template>
	<div>
        <!-- 固定写法$t(调用js文件里变量为m的数据) -->
        <p>{{ $t('m.index') }}</p>
        <div>
			<!-- 开关标签 -->
			<h-switch @change="lang_change" v-model="lang">中/英</h-switch>
		</div>
    </div>
</template>
<script>
    export default {
        data () {
            return {
                // 开关变量,1默认为True
                lang:1,
            }
        },
        // 钩子方法
        mounted:function(){
            var lang_locale = localStorage.getItem('lang');
            // 判断用户是否设置过语言
            if (lang_locale){
                this.$i18n.locale = lang_locale;
                // 手动赋值变量lang
                if (lang_locale == 'zh'){
                    this.lang = 0;
                }else{
                    this.lang = 1;
                }
             // 如果没有登陆默认设置为中文
            }else{
                this.$i18n.locale = 'zh';
                this.lang = 1;
            }
        },
        methods:{
            // 切换语言方法
            lang_change:function(){
                // 判断lang变量是否为True
                if (this.lang == true){
                    this.$i18n.locale = 'en';
                    localStorage.setItem('lang','en');
                }else{
                    this.$i18n.locale = 'zh';
                    localStorage.setItem('lang','zh');
                }
            }
        },
    }
</script>
