滑块验证码

# 首先vue中得安装好 vue-drag-verify

$ npm install vue-drag-verify --save

<template>
	<div>
         <!-- 滑块验证码 -->
		<drag-verify class="verifycode" :width="width" :height="height" :text="text" ref="Verify">		  </drag-verify>

        <Button @click="submit" color="blue">提交</Button>
    </div>
</template>
<script>
	// 导入滑块验证码
	import dragVerify from 'vue-drag-verify'
    export default {
        data(){
            return{
            // 滑块验证码相关数据
            width:170,
            height:30,
            text:'向右拉动>>',
            }
        },
        components:{
            // 引入组件
            'dragVerify':dragVerify
        },
        methods:{
            submit:function(){
            // 判断是否拖动
            console.log(this.$refs.Verify.isPassing);
            if(this.$refs.Verify.isPassing == false){
                this.$Message('滑动验证失败');
                return false;
            }
            }
        },
    }
</script>
