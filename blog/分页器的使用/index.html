分页逻辑流程图

后端(django)，将数据切割

from rest_framework.views import APIView
from myapp.models import Goods
from myapp.myser import GoodsSer
from rest_framework.response import Response

# 商品列表页

class GoodsList(APIView):
    def get(self,request):
        # 当前页

        page = int(request.GET.get('page',1))
        # 一页的数据个数

        size = int(request.GET.get('size',4))

        # 计算从什么位置开始切(当前页-1,再*一页的数据个数)

        data_start = (page-1) * size

        # 计算切到什么位置结束

        data_end = page * size

        # 查询 切片操作(已切完的数据)

        goods = Goods.objects.all()[data_start:data_end]

        # 查询所有商品个数

        count = Goods.objects.count()

        # 序列化

        goods_ser = GoodsSer(goods,many=True)

        return Response({'data':goods_ser.data,'total':count})

前端(vue.js)

<template>
	<div>
        <div v-for="(item,index) in goodslist" :key="index">
            <video width="200" height="150" :src="item.video" controls="controls"></video>
            <h4><a :href="'http://localhost:8080/item/?id='+item.id">{{ item.name }}</a></h4>
            <p><span class="emphasis">{{ item.price }}￥</span></p>
        </div>
        <!-- 自带分页器(如果想改样式，在标签内加layout属性：layout="pager,jumper,total"等) -->
        <div>
            <Pagination @change="get_goods" v-model="pagination"></Pagination>
        </div>
    </div>
</template>
<script>
    export default {
        data () {
            return {
                // 分页器变量
                pagination:{
                    // 当前页
                    page:1,
                    // 每页的数据量
                    size:2,
                    // 总数据
                    total:0,
                },
                // 商品列表
                goodslist:[],
            }
        },
        mounted:function(){
            // 商品展示
            this.get_goods();
        },
        methods:{
            // 获取商品信息
            get_goods:function(){
                this.axios.get('http://127.0.0.1:8000/goodslist/',{params:{page:this.pagination.page,size:this.pagination.size}}).then((result)=>{
                    this.goodslist = result.data.data;
                    this.pagination.total = result.data.total;
                });
            }
        },
    }
</script>
