后端见(分页器的使用.md)

前端(vue.js)demo

<template>
	<div>
        
        <div v-for="(item,index) in goodslist" :key="index">
            <video width="200" height="150" :src="item.video" controls="controls"></video>
            <h4><a :href="'http://localhost:8080/item/?id='+item.id">{{ item.name }}</a></h4>
            <p><span class="emphasis">{{ item.price }}￥</span></p>
        </div>
        <!-- 自定义分页器 -->
        <div>
            <a @click="get_goods_self(1)">首页</a>&emsp;&emsp;
            <!-- 上一页 -->
            <Button v-show="lastpage" @click="get_goods_self(lastpage)">上一页</Button>&emsp;&emsp;
            <span v-for="item in last_page">
                <a @click="get_goods_self(item)">{{ item }}</a>&emsp;&emsp;
            </span>
            <!-- 当前页 -->	
            <a @click="get_goods_self(page)">{{ page }}</a>&emsp;&emsp;
            <!-- 右偏移 -->
            <span v-for="item in next_page">
                <a @click="get_goods_self(item)">{{ item }}</a>&emsp;&emsp;
            </span>
            <!-- 下一页 -->
            <Button v-show="nextpage" @click="get_goods_self(nextpage)">下一页</Button>&emsp;&emsp;
            <a @click="get_goods_self(allpage)">尾页</a>
        </div>
    </div>
</template>
<script>
    export default {
        data () {
            return {
                // 分页偏移
                last_page:[],  // 左
                next_page:[],  // 右
                // 自定义分页商品列表
                goodslist_self:[],
                // 自定义分页商品总数
                total_self:0,
                // 上一页
                lastpage:0,
                // 下一页
                nextpage:0,
                // 当前页
                page:1,
                // 每页展示数量
                size:4,
                // 总页数
                allpage:0,
            }
        },
        mounted:function(){
            this.get_goods_self(1);
        },
        methods:{
            // 自定义分页获取商品信息
            get_goods_self:function(page){
                this.page = page;
                this.axios.get('http://127.0.0.1:8000/goodslist/',{params:{page:page,size:this.size}}).then((result)=>{
                    // 数据赋值
                    this.goodslist_self = result.data.data;
                    // 商品总数
                    this.total_self = result.data.total;
                    // 判断上一页
                    if (page == 1){
                        this.lastpage = 0;
                    }else{
                        this.lastpage = page - 1;
                    }
                    // 计算总页数(Math.ceil)向上取整
                    this.allpage = Math.ceil(this.total_self / this.size);
                    // 判断下一页
                    if (page == this.allpage){
                        this.nextpage = 0;
                    }else{
                        this.nextpage = page + 1;
                    }
                    // 设置偏移量
                    var move_page = 2;
                    // 计算左侧偏移量
                    var my_last = [];
                    for (let i=page-move_page;i<page;i++){
                        if (i > 0){
                            my_last.push(i);
                        }
                    }
                    // 计算右偏移量
                    var my_next = [];
                    for (let i=page+1;i<=page+move_page;i++){
                        if (i <= this.allpage){
                            my_next.push(i);
                        }
                    }
                    this.last_page = my_last;
                    this.next_page = my_next;
                })
            }
        }
    }
    
</script>
