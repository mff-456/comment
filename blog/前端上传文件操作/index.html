    <template>
    <div>
        <myheader></myheader>
   
        <section class="featured-block text-center">
            <div class="container">
                <div>
                    <Breadcrumb :datas="datas"></Breadcrumb>
                </div>
            </div>

        <div>
            <Avatar :src="src" :width="100" fit="fill"></Avatar>
        </div>
        <div>
            <table>
                <tr>
                    <td>
                        {{ username }}&nbsp;&nbsp;的用户头像：
                    </td>
                    <td>
                        <input type="file" @change="upload">
                        <button @click="submit" v-show="type">提交</button>                    
                    </td>
                </tr>
            </table>
        </div>
        <footer class="footer">
            <div class="container">
                @v3u.cn
            </div>

<script>
// 导入组件
import myheader from './myheader.vue'
   
export default {
    data(){
        return{
            msg:"这是一个变量",
            username:'',
            src:'',
            type:0,
            user:'',
            // 面包屑导航变量
            datas:[{title:'首页',route:{name:'index'}},{title:'用户中心->我的首页'}]
        }
    },
    components:{
        'myheader':myheader,
    },
    // 钩子
    mounted:function(){
        this.username = this.$route.query.username;
   
    },
    methods:{
          
        upload:function(e){
            // 获取文件
            let file = e.target.files[0];
            // 声明表单参数
            let param = new FormData();
               
            // 文件添加到param
            param.append('file',file,file.name);
            // 声明请求头
            let config = {headers:{'Content-Type':'multipart/form-data'}}
   
            // 请求后台接口
            this.axios.post("http://127.0.0.1:8000/upload/",param,config).then((result)=>{
                console.log(result);
                // 拿出图片赋值给变量src
                this.src = 'http://127.0.0.1:8000/static/upload/' + result.data.filename;
                if(this.src){
                    this.type = 1;
                }
   
            });
        },
         // 点击提交将图片链接存储到数据库   
        submit:function(){
            // 提交图片向后端返回数据
            this.axios.get("http://127.0.0.1:8000/userimgs/",{params:{username:this.username,src:this.src}}).then((result)=>{
                console.log(result);
                localStorage.setItem("imgs",this.src);
                this.$router.push('/')
                this.$Message(result.data.message);
                   
            })
        }
    }
   
   
}
</script>

  2. 这里需要在后端(django)中做一个简单的存储图片链接的接口(python文件)

    # 用户头像保存

class UserImgs(APIView):
    def get(self,request):
        # 接收图片链接

        img_link = request.GET.get('src',None)
   
        # 接收用户名信息

        username = request.GET.get('username',None)
   
        User.objects.filter(username=username).update(img=img_link)
           
        return Response({'src':img_link,'code':200,'message':'头像上传成功'})

 data () {

       console.log(result);
       this.src = result.data.src;
   		  
   
   })
   // 接收微博用户信息
   var sina_name = this.$route.query.sina_name;
   var user_id = this.$route.query.uid;
   // 判断路由是否有用户信息(正常用户请求路由没有携带路由参数)
   if(user_id){
       localStorage.setItem("username",sina_name);
       localStorage.setItem("uid",user_id);
       localStorage.setItem('imgs',this.src);
   }
   
   //判断是否登录
   var uname = localStorage.getItem("username");
   if(uname==null){
       //没登录
       this.username = '';
   }else{
     this.username = uname;
     this.imgs = localStorage.getItem('imgs')
   }
   
  },
  methods:{
   submit:function(){
       this.$Message('退出成功');
       localStorage.removeItem('username');
       localStorage.removeItem('uid');
       this.$router.push('/login');
   },
        
  }
}
</script>
