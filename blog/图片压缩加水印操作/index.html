图片的压缩及加水印

  1. 图片加水印

    from PIL import Image,ImageDraw,ImageFont
# 打开一张图片

im = Image.open("./mff.jpg")
# 打开一张图片

im = Image.open("./mff.jpg")
   
# im.format图片类型  im.size图片的宽高  im.mode颜色模式

print(im.format,im.size,im.mode)
   
# 生成画笔

draw = ImageDraw.Draw(im)
   
# 设置字体

myfont = ImageFont.truetype('C:/windows/fonts/Arial.ttf', size=80)
   
# 绘制水印  图片位置--水印内容--fill:设置文字透明度--字体大小

draw.text((3840-130,2160-100),'mff',fill = (249, 249, 249),font=myfont)
   
# 查看

im.show()
# 保存

im.save('./static/upload/mff.jpg')

  2. 图片压缩

    import cv2
# 读图

img = cv2.imread('./static/upload/mff.jpg')
   
# 压缩(png格式照片则压成jpg格式)

cv2.imwrite('./static/upload/mff.jpg',img,[cv2.IMWRITE_JPEG_QUALITY,0])
