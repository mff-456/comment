python中的装饰器简单来说就是在不改变原代码的情况下，对其进行附加一些功能(装饰器只会在请求之前做响应操作)

demo(拿jwt权限验证来做演示)

# 定义权限检测装饰器

def my_decorator(func):
    def wrapper(request,*args,**kwargs):
        uid = request.GET.get('uid',None)
        myjwt = request.GET.get('jwt',None)
        # 验证用户合法性

        try:
            decode_jwt = jwt.decode(myjwt,'mff-project',algorithms=['HS256'])
            print(decode_jwt['data']['uid'])
        except Exception as e:
            print(e)
            return Response({'code':403,'message':'无权访问'})

        # 进行比对

        if int(uid) != int(decode_jwt['data']['uid']):

            return Response({'code':401,'message':'密钥无权限'})
		
        return func(request,*args,**kwargs)
	# 返回内部函数

    return wrapper

调用装饰器

# 类视图加装饰器包

from django.utils.decorators import method_decorator

# 测试jwt接口

class Jwt_test(APIView):
    # 这里在类视图中使用method_decorator

    @method_decorator(my_decorator)
    # 如果在方法视图中直接用装饰器名称即可(my_decorator)

    def get(self,request):
        uid = request.GET.get('uid',None)
        user = User.objects.get(id=int(uid))

        return Response({'uid':uid,'username':user.username,'code':200,'message':'成功'})
