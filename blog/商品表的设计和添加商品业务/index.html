设计图

当商品存为单表时，新增一种商品类别就得新建一张表，上万种商品类就得建上万张表，虽然能提供良好的检索服务，但使很难维护

而右边图将所有商品以json形式存储到一张表里，易维护，但是检索服务比较繁琐，这也是唯一的缺点

建立一对多关系模型

from django.db import models
# 导入时间域

from django.utils import timezone

# 基类

class Base(models.Model):
    #时间字段

    create_time = models.DateTimeField(default=timezone.now,null=True)

    class Meta:
        abstract = True
# 商品表

class Goods(Base):
    # 名称

    name = models.CharField(max_length=200)
    # 商品描述

    desc = models.CharField(max_length=200,null=True)
    # 图片

    img = models.CharField(max_length=200)
    # 展示视频

    video = models.CharField(max_length=200,null=True)
    # 价格

    price = models.IntegerField()
    # 规格

    params = models.CharField(max_length=400)
    # 关注数

    flows = models.IntegerField(default=0,null=True)
    # 外键关系(这里的外键可以理解为逻辑外键，可以实现外键的功能，但不会受到外键的约束)

    cid = models.IntegerField(null=True)


    # 声明表名

    class Meta:
        db_table = "goods"

# 商品类表

class Category(Base):
    name = models.CharField(max_length=200,null=True)

        # 声明表名

    class Meta:
        db_table = "category"

执行迁移命令生成数据表

新建序列化操作文件(myser.py)

#导包

from rest_framework import serializers

# 导入需要序列化的表

from myapp.models import Goods,Category

class GoodsSer(serializers.ModelSerializer):

    class Meta:
        model = Goods
        fields = "__all__"
        
class CategorySer(serializers.ModelSerializer):

    class Meta:
        model = Category
        fields = "__all__"

新建商品(goods.py)文件

from rest_framework.views import APIView
from myapp.models import Goods,Category
from myapp.myser import GoodsSer,CategorySer
from rest_framework.response import Response

# 商品分类

class CategoryList(APIView):
    def get(self,request):
        
        category = Category.objects.all()

        category_ser = CategorySer(category,many=True)

        return Response(category_ser.data)
    
    
# 商品入库

class InsertGoods(APIView):
    def get(self,requests):
        # 接参

        name = requests.GET.get('name',None)
        price = requests.GET.get('price',None)
        # 这里接收数据为数组，以字符串的方式进行存储

        params = requests.GET.get('param',None)
        img = requests.GET.get('img',None)
        video = requests.GET.get('video',None)
        selected = requests.GET.get('selected',None)

        # 排重操作

        goods = Goods.objects.filter(name=name).first()

        if goods:

            return Response({'code':403,"message":"您以添加过商品"})

        # 查询商品分类id

        category = Category.objects.filter(name=selected).first()

        # 入库

        goods = Goods(name=name,price=price,params=params,img=img,cid=category.id,video=video)

        goods.save()
        
        return Response({'code':200,"message":"添加成功"})

这里是一个添加分类和商品的接口，记得配好接口路由

新建添加商品(addgoods.vue)文件

<template>
    <div>
        <myheader></myheader>

        <section class="featured-block text-center">
            <div class="container">
                <div>
                    <Breadcrumb :datas="datas"></Breadcrumb>
                </div>
            </div>
        </section>
        
        <div>
            <table>
                <tr>
                    <td>
                        商品分类：
                    </td>
                    <td>
                        <Select v-model="selected" :datas="category"></Select>
                    </td>
                </tr>
                 <tr>
                    <td>
                        商品名称：
                    </td>
                    <td>
                        <input type="text" v-model="name" placeholder="请输入商品名称">
                    </td>
                </tr>
                <tr>
                    <td>
                        商品价格：
                    </td>
                    <td>
                        <input type="number" v-model="price" placeholder=0>
                    </td>
                </tr>
                <tr>
                    <td>
                        颜色：
                    </td>
                    <td>
                        <input type="text" v-model="color" placeholder="请输入商品颜色类">
                    </td>
                </tr>
                <tr>
                    <td>
                        尺寸：
                    </td>
                    <td>
                        <input type="text" v-model="size" placeholder="请输入商品尺寸">
                    </td>
                </tr>
                <tr>
                    <td>
                        季节：
                    </td>
                    <td>
                        <input type="text" v-model="season" placeholder="请输入季节类">
                    </td>
                </tr>
                <tr>
                    <td>
                        上传商品图片：
                    </td>
                    <td>
                        <input type="file" @change="upload_qiniu">
                        <br>
                        {{ load_percent }}
                        <!-- 进度条标签 -->
                        <Progress :percent="load_int" color="green" v-show="load_int">						</Progress>
                    </td>
                </tr>
                <tr>
                    <td>
                        展示视频：
                    </td>
                    <td>
                        <input type="file" @change="upload_qiniu">
                        <br>
                        {{ load_percent }}
                        <Progress :percent="load_int" color="green" v-show="load_int">						</Progress>
                    </td>
                </tr>
                <tr>
                    <td></td>
                    <td>
                        <button @click="submit">确定上传</button>
                    </td>
                </tr>
            </table>
        </div>
        <footer class="footer">
            <div class="container">
                @v3u.cn
            </div>
        </footer>
    </div>
  
</template>
<script>
// 导入组件
import myheader from './myheader.vue'
// 导入文件
import {config,formatXml} from '../config.js'

export default {
    data(){
        return{
            // 下拉框默认
            selected:'衣服',
            // 商品分类
            category:[],
            // 上传进度展示
            load_percent:'',
            // 整型进度
            load_int : 0,
            // 视频链接
            video:'',
            // 图片链接
            src:'', 
            // 七牛云token
            token:'',
            name:'',
            price:'',
            color:'',
            size:'',
            season:'',
            // 面包屑导航变量
            datas:[{title:'首页',route:{name:'index'}},{title:'添加商品'}]
        }
    },
    components:{
        'myheader':myheader,
    },
    // 钩子
    mounted:function(){
        this.get_token();
        this.get_category();

    },
    methods:{
        // 获取商品分类数据
        get_category:function(){
            
            this.axios.get('http://127.0.0.1:8000/categorylist/').then((result)=>{
                
                console.log(result);
                var mycate = [];
                // 二次处理数据
                for (let i=0;i<result.data.length;i++){
                    mycate.push(result.data[i]['name']);
                };
                // 赋值
                this.category = mycate;
            });


        },
        // 获取七牛云token   
        get_token:function(){
            // 请求后台接口
            this.axios.get("http://127.0.0.1:8000/qiniu/").then((result)=>{
                console.log(result);
                // 赋值给token
                this.token = result.data.token;
                console.log(this.token);
            });
        },
        // 上传文件
        upload_qiniu:function(e){
            var size = 0;
            // 获取文件
            let file = e.target.files[0];
            size = file.size;
            size = size / 1024;
            size = (size / 1024).toFixed(3);
            // 声明表单参数
            let param = new FormData();
            // 文件添加到param
            param.append('file',file,file.name);
            // 添加token
            param.append('token',this.token);
            // 自定义axios
            const axios_qiniu = this.axios.create({withCredentials:false});
            // 发送请求
            axios_qiniu({
                method:'POST',
                url:'http://up-z1.qiniu.com/',
                data:param,
                timeout:30000,
                // 上传过程中的方法
                onUploadProgress:(e)=>{
                    // 计算上传百分比
                    var complete = (e.loaded / e.total);
                    // 处理美化
                    if(complete < 1){
                        // (complete为以完成的量 * 100).保留两位小数 + '%'
                        this.load_percent  = (complete * 100).toFixed(2) + '%';
                        // 整型进度
                        this.load_int = parseInt((complete * 100).toFixed(2));
                    };
                }                
            }).then((result)=>{
                // 手动赋值%100(防止用户上传完成compltet还在计算)
                this.load_percent = '100%';
                this.load_int = 100;
                console.log(result.data);
                // 这里判断用户的文件是否大于4m
                if (size<4){
                    this.src = 'http://q9xyoh7sx.bkt.clouddn.com/' + result.data.key;
                }else{
                    this.video = 'http://q9xyoh7sx.bkt.clouddn.com/' + result.data.key
                }
            })
        },
        // 添加商品
        submit:function(){
            console.log(this.selected);
            // 将普通字段转换为json
            var param = {};

            param['color'] = this.color;
            param['size'] = this.size;
            param['season'] = this.season;
            // 转换为字符串
            param = JSON.stringify(param);
            console.log(param);
            this.axios.get('http://127.0.0.1:8000/insertgoods/',{params:{name:this.name,price:this.price,img:this.src,param:param,selected:this.selected,video:this.video}}).then((result)=>{

                console.log(result);

                this.$Message(result.data.message);
            })
        }     
    }
}
</script>

这里需要注意的是，商品规格是为数组，需将数组拆分为对应字段供用户添加商品使用，当用户点击提交时，将拆分的数据转化为json格式，再将json转换为字符串进行存储

上传文件接口为七牛云上传文件接口，具体参照（七牛云进度条显示上传文件.md）

头部组件参照(实现模拟注册登录.md)进行复制
